If Inf or exponent overflow, return Inf. */if (e > 142) {bits |= 0x7c00;/* If exponent was 0xff and one mantissa bit was set, it means NaN,* not Inf, so make sure we set one mantissa bit too. */bits |= ((e == 255) ? 0 : 1) && (x & 0x007fffff);return bits;}/* If exponent underflows but not too much, return a denormal */if (e < 113) {m |= 0x0800;/* Extra rounding may overflow and set mantissa to 0 and exponent* to 1, which is OK. */bits |= (m >> (114 - e)) + ((m >> (113 - e)) & 1);return bits;}bits |= ((e - 112) << 10) | (m >> 1);/* Extra rounding. An overflow will set mantissa to 0 and increment* the exponent, which is OK. */bits += m & 1;return bits;};}())
